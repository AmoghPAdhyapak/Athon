here is the code : Architectural Root Cause & Separation Strategy
The system breakdown stems from a classical shared context leakage. During the rapid parallel integration of the Platform Subtype Intelligence Matrix and the Scene Decoder Engine, state machines and mutation pipelines inside the generation hooks became unified under an abstract single-generation handler. Consequently, the text workbench in Create.tsx began tracking global file and image buffer states, matching conditions meant exclusively for multimodal image synthesizers, and issuing runtime queries to /api/analyze/* endpoints instead of the streaming text matrix pipeline.

To enforce permanent, hard-walled architectural separation, we decouple the codebase into two entirely distinct operational layers:

The Textual Synthesis Engine (src/pages/Create.tsx): Locked entirely to text/string manipulation payloads. It orchestrates state via localized arrays, controls platform-aware content drops, and targets exclusive streaming endpoints (/api/captions/*). It contains absolute zero image uploading or thumbnail generation logic.

The Media Production Workspace (src/pages/Analyze.tsx or ThumbnailGeneration.tsx): Isolated exclusively for processing image buffers, coordinate arrays, multipart/form-data payloads, and deep multi-modal Scene Decoder interpretation models via the visual endpoints (/api/analyze/*).

import { useState, useEffect } from "react";
import { StudioLayout } from "@/components/layout/WorkspaceLayouts";
import { Button } from "@/components/ui/button";

import { useToast } from "@/hooks/use-toast";

type PlatformType = "Instagram" | "YouTube" | "TikTok" | "Facebook";

const SUBTYPE_MATRIX: Record<PlatformType, string[]> = {
  Instagram: ["Reel", "Story", "Post", "Carousel"],
  YouTube: ["Long Video", "Shorts", "Community Post"],
  TikTok: ["Viral Clip", "Story Style", "Trend Hook"],
  Facebook: ["Post", "Story", "Reel"]
};

export default function Create() {
  const { toast } = useToast();
  
  // Localized Text/Platform State Contexts Only - Absolute Separation from Image State Loops
  const [prompt, setPrompt] = useState("");
  const [selectedPlatform, setSelectedPlatform] = useState<PlatformType>("Instagram");
  const [selectedSubtype, setSelectedSubtype] = useState("Reel");
  const [tonalTrack, setTonalTrack] = useState("Cinematic Focus");
  
  // Streaming Compilation States
  const [outputStream, setOutputStream] = useState("");
  const [isCompiling, setIsCompiling] = useState(false);
  const [hasCopied, setHasCopied] = useState(false);

  // Synchronize structural subtypes matching platform state transitions
  useEffect(() => {
    setSelectedSubtype(SUBTYPE_MATRIX[selectedPlatform][0]);
  }, [selectedPlatform]);

  // Safeguarded text-handoff capture layer from Scene Decoder cache hooks
  useEffect(() => {
    const cachedPrompt = sessionStorage.getItem("creatorcore:staged_prompt");
    if (cachedPrompt) {
      setPrompt(cachedPrompt);
      sessionStorage.removeItem("creatorcore:staged_prompt");
      toast({ 
        title: "Context Injection Successful", 
        description: "Visual analytical description mounted cleanly into text prompt matrix." 
      });
    }
  }, [toast]);

  const executeTextPipelineStream = async () => {
    if (!prompt.trim()) return;
    
    setIsCompiling(true);
    setOutputStream("");
    
    try {
      // Direct hard-wired endpoint execution targeting ONLY the text matrix generation routers
      const response = await fetch("/api/captions/generate-platform-subtype", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          platform: selectedPlatform,
          subtype: selectedSubtype,
          prompt,
          tonalTrack
        })
      });

      if (!response.body) throw new Error("Text response streaming layout channel failed to resolve.");
      
      const reader = response.body.getReader();
      const decoder = new TextDecoder();
      let finished = false;

      while (!finished) {
        const { value, done } = await reader.read();
        finished = done;
        const chunkText = decoder.decode(value, { stream: !done });
        
        const lines = chunkText.split("\n");
        for (const line of lines) {
          if (line.startsWith("data: ")) {
            const rawData = line.slice(6).trim();
            if (rawData === "[DONE]") break;
            try {
              const parsed = JSON.parse(rawData);
              if (parsed.text) setOutputStream((prev) => prev + parsed.text);
            } catch (e) {
              // Gracefully handle segmented buffer parsing anomalies
            }
          }
        }
      }
    } catch (err) {
      toast({ 
        variant: "destructive", 
        title: "Text Compilation Engine Failure", 
        description: "Streaming transport error during script/caption compilation orchestration." 
      });
    } finally {
      setIsCompiling(false);
    }
  };

  const copyToClipboard = () => {
    if (!outputStream) return;
    navigator.clipboard.writeText(outputStream);
    setHasCopied(true);
    setTimeout(() => setHasCopied(false), 2000);
    toast({ title: "Buffer Staged", description: "Text output copied to system clipboard storage." });
  };

  return (
    <StudioLayout 
      title="AI Creator Text Synthesis Workspace" 
      description="Deploy high-retention copy, headlines, or metadata optimized across platform subtype algorithmic patterns."
    >
      {/* CONTROL INTERFACE COLUMNS - 100% ISOLATED FROM IMAGE PAYLOADS */}
      <div className="xl:col-span-5 border-r border-border/40 bg-card/10 flex flex-col h-full overflow-hidden shrink-0">
        <div className="flex-1 overflow-y-auto p-5 space-y-6">
          
          {/* Platform Selector Field */}
          <div className="space-y-2.5">
            <label className="text-[11px] font-mono uppercase tracking-wider text-muted-foreground font-bold flex items-center gap-2">
              <Radio className="w-3.5 h-3.5 text-primary" /> Target Channels
            </label>
            <RadioGroup 
              value={selectedPlatform} 
              onValueChange={(val) => setSelectedPlatform(val as PlatformType)}
              className="grid grid-cols-4 gap-2 p-1 bg-muted/30 border border-border/40 rounded-xl"
            >
              {(Object.keys(SUBTYPE_MATRIX) as PlatformType[]).map((plat) => (
                <div key={plat} className="flex items-center justify-center">
                  <RadioGroupItem value={plat} id={`plat-${plat}`} className="sr-only" />
                  <Label
                    htmlFor={`plat-${plat}`}
                    className={`w-full text-center text-xs font-medium py-2 rounded-lg cursor-pointer transition-all border ${
                      selectedPlatform === plat 
                        ? "bg-background border-border/80 text-foreground shadow-xs" 
                        : "border-transparent text-muted-foreground hover:text-foreground"
                      }`}
                  >
                    {plat}
                  </Label>
                </div>
              ))}
            </RadioGroup>
          </div>

          {/* Platform Variant Subtype Blocks */}
          <div className="space-y-2.5">
            <label className="text-[11px] font-mono uppercase tracking-wider text-primary font-bold flex items-center gap-2">
              <Layers className="w-3.5 h-3.5" /> Contextual Target Subtype
            </label>
            <div className="grid grid-cols-2 gap-2">
              {SUBTYPE_MATRIX[selectedPlatform].map((sub) => (
                <button
                  key={sub}
                  type="button"
                  onClick={() => setSelectedSubtype(sub)}
                  className={`text-xs p-3 rounded-xl border text-left flex flex-col justify-between transition-all h-16 ${
                    selectedSubtype === sub
                      ? "bg-primary/5 border-primary text-foreground shadow-[0_0_12px_rgba(124,58,237,0.12)]"
                      : "bg-background/40 border-border/50 text-muted-foreground hover:border-border hover:text-foreground"
                  }`}
                >
                  <span className="font-semibold text-foreground text-xs">{sub}</span>
                  <span className="text-[9px] font-mono text-muted-foreground/60 uppercase">ALGO MATRIX MOUNTED</span>
                </button>
              ))}
            </div>
          </div>

          {/* Master Text Prompt Directives Component */}
          <div className="space-y-2">
            <label className="text-[11px] font-mono uppercase tracking-wider text-muted-foreground font-bold flex items-center gap-1.5">
              <FileText className="w-3.5 h-3.5" /> Structural Copy Brief
            </label>
            <textarea
              value={prompt}
              onChange={(e) => setPrompt(e.target.value)}
              className="w-full h-36 bg-background/40 border border-border/50 focus:border-primary rounded-xl p-3 text-sm text-foreground focus:outline-hidden transition-colors resize-none placeholder:text-muted-foreground/40 leading-relaxed font-sans"
              placeholder="State topic strings, hook contexts, target audience profiles, or structural directives..."
            />
          </div>

          {/* Tonal Variant Blocks */}
          <div className="space-y-2">
            <label className="text-[11px] font-mono uppercase tracking-wider text-muted-foreground font-bold">
              Algorithmic Tone Configuration
            </label>
            <div className="grid grid-cols-2 gap-2">
              {["Curiosity Loop", "Cinematic Focus", "Retention Heavy", "SEO Optimized"].map((tone) => (
                <button
                  key={tone}
                  type="button"
                  onClick={() => setTonalTrack(tone)}
                  className={`text-xs py-2 px-3 rounded-lg border font-medium text-center transition-all ${
                    tonalTrack === tone
                      ? "bg-foreground text-background border-foreground font-semibold shadow-md"
                      : "bg-background/30 border-border/50 text-muted-foreground hover:text-foreground hover:border-border"
                  }`}
                >
                  {tone}
                </button>
              ))}
            </div>
          </div>

        </div>

        {/* Text Trigger Submission Footbar Deck */}
        <div className="p-4 border-t border-border/40 bg-background/50 shrink-0">
          <Button
            onClick={executeTextPipelineStream}
            disabled={isCompiling || !prompt.trim()}
            className="w-full h-11 text-xs font-mono font-bold tracking-widest uppercase gap-2 bg-primary hover:bg-primary/90 text-white"
          >
            {isCompiling ? (
              <RefreshCw className="w-4 h-4 animate-spin" />
            ) : (
              <Sparkles className="w-4 h-4 fill-current" />
            )}
            COMPILE SYSTEM TEXTPAYLOAD
          </Button>
        </div>
      </div>

      {/* RIGHT PREVIEW DOCK MONITOR VIEWPORT: ONLY TRACKS TEXT STREAM EMISSIONS */}
      <div className="xl:col-span-7 bg-black/10 dark:bg-neutral-950/20 p-6 flex flex-col h-full overflow-hidden">
        <div className="flex items-center justify-between border-b border-border/30 pb-3 mb-4 shrink-0">
          <span className="text-xs font-mono tracking-wider text-muted-foreground uppercase flex items-center gap-2">
            <span className={`h-1.5 w-1.5 rounded-full ${isCompiling ? "bg-amber-400 animate-pulse" : "bg-primary"}`} />

    if (!file) return;

    setFileMimeType(file.type);

    reader.onload = () => {
      if (typeof reader.result === "string") {

        setTelemetryReport(null); // Clear cached data objects

    reader.readAsDataURL(file);
  };

        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ 
          imageBase64: imageFileBase64, 
          mimeType: fileMimeType

        variant: "destructive",

              Image Stream Interface
