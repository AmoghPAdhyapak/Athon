export default function SceneDecoder() {

  const [base64Data, setBase64Data] = useState<string | null>(null);

        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ imageBase64: base64Data, mimeType })
      });

      if (!response.ok) throw new Error("Telemetry analysis transmission failure.");

      toast({ title: "Scene Analysis Synthesized", description: "Cinematic structural metrics successfully unboxed." });
    } catch (err) {
      toast({ variant: "destructive", title: "Diagnostic Failure", description: "AI image interpreter pipeline aborted due to payload loss." });
    } finally {
      setIsDecoding(false);
    }
  };

  const stageForWorkflow = () => {
    if (!telemetry) return;
    // Inject metadata variables directly into session persistence memory layout
    sessionStorage.setItem("creatorcore:staged_prompt", telemetry.creatorReadyPrompt);
    toast({ title: "Handoff Locked", description: "Visual parameters copied to text creation workspace workflow." });
    setLocation("/create");
  };

  const copyStringField = (fieldKey: string, text: string) => {
    navigator.clipboard.writeText(text);
    setCopiedField(fieldKey);
    setTimeout(() => setCopiedField(null), 1500);
  };

  return (
    <SceneDecoderLayout title="AI Cinematic Visual Interpreter" statusText={isDecoding ? "DECODING BLOCK" : "MONITOR READY"}>
      
      {/* LEFT COMPONENT PANE: ASSET LOADING PANEL & CANVAS FRAME (5-COLUMNS) */}
      <div className="xl:col-span-5 flex flex-col gap-6 h-full overflow-y-auto pr-2">
        <Card className="bg-card/20 border-border/40 backdrop-blur-md">
          <CardContent className="p-5 space-y-4">
            <div>
              <h2 className="text-sm font-semibold tracking-tight text-foreground">Media Core Processing</h2>
              <p className="text-xs text-muted-foreground mt-0.5">Stage raw video source frame to perform deep visual parameters extraction.</p>
            </div>

            {/* Interactive Upload Canvas Frame Track */}
            <div className="relative aspect-video rounded-xl border-2 border-dashed border-border/60 hover:border-cyan-500/40 bg-background/50 flex flex-col items-center justify-center overflow-hidden transition-all group p-2">
              {base64Data ? (
                <>
                  <img 
                    src={`data:${mimeType};base64,${base64Data}`} 
                    alt="Active decoding target" 
                    className="w-full h-full object-cover rounded-lg"
                  />
                  <div className="absolute inset-0 bg-background/60 opacity-0 group-hover:opacity-100 flex items-center justify-center transition-all duration-200">
                    <label className="cursor-pointer bg-background border border-border/80 px-4 py-2 rounded-lg text-xs font-mono text-foreground font-semibold">
                      REPLACE SOURCE ASSET
                      <input type="file" accept="image/*" onChange={processFileBuffer} className="hidden" />
                    </label>
                  </div>
                </>
              ) : (
                <label className="cursor-pointer flex flex-col items-center justify-center w-full h-full py-10">
                  <div className="h-10 w-10 rounded-xl bg-muted/60 border border-border/40 flex items-center justify-center group-hover:scale-105 transition-transform">
                    <UploadCloud className="w-4 h-4 text-muted-foreground" />
                  </div>
                  <span className="text-xs font-medium mt-3 text-foreground tracking-tight">Ingest Production Source Frame</span>
                  <span className="text-[10px] font-mono text-muted-foreground/60 mt-1 uppercase">JPEG, PNG, WEBP UP TO 12MB</span>
                  <input type="file" accept="image/*" onChange={processFileBuffer} className="hidden" />
                </label>
              )}
            </div>

            <Button
              onClick={executeDecodeAnalysis}
              disabled={isDecoding || !base64Data}
              className="w-full h-11 font-mono text-xs font-bold tracking-widest uppercase gap-2 bg-gradient-to-r from-cyan-600 to-blue-600 hover:from-cyan-500 hover:to-blue-500 text-white"
            >
              {isDecoding ? <RefreshCw className="w-4 h-4 animate-spin" /> : <Eye className="w-4 h-4" />}
              RUN CINEMATIC SCENE DECODE
            </Button>
          </CardContent>
        </Card>

        {/* Action Handoff Routing Card */}
        {telemetry && (
          <Card className="bg-gradient-to-br from-cyan-950/20 to-background border-cyan-500/20 shadow-lg animate-fade-in">
            <CardContent className="p-5 space-y-3">
              <div className="flex items-center gap-2">
                <Sparkles className="w-4 h-4 text-cyan-400 fill-current" />
                <h4 className="text-xs font-mono uppercase tracking-wider text-cyan-400 font-bold">Ecosystem Workflow Engine</h4>
              </div>
              <p className="text-xs text-muted-foreground leading-normal">
                Visual prompts match target structures accurately. Push telemetry metrics into the platform content generation views instantly.
              </p>
              <Button onClick={stageForWorkflow} className="w-full h-10 gap-1.5 text-xs font-mono font-bold tracking-wide uppercase">
                FORWARD TO STUDIO WORKSPACE <ArrowUpRight className="w-4 h-4" />
              </Button>
            </CardContent>
          </Card>
        )}
      </div>

      {/* RIGHT COMPONENT PANE: METRICS TELEMETRY REPORT STREAM PANELS (7-COLUMNS) */}
      <div className="xl:col-span-7 flex flex-col h-full overflow-hidden">
        <div className="flex-1 bg-card/10 border border-border/40 rounded-xl p-5 md:p-6 overflow-y-auto space-y-6 relative backdrop-blur-md">
          {telemetry ? (
            <div className="space-y-6 animate-fade-in">
              
              {/* Parameter Row 1: Direct Core Prompts */}
              <div className="bg-background/40 border border-border/30 rounded-xl p-4 space-y-2 relative group">
                <div className="flex justify-between items-center">
                  <span className="text-[10px] font-mono tracking-widest uppercase text-cyan-400 font-bold flex items-center gap-1.5">
                    <Sparkles className="w-3.5 h-3.5" /> Creator-Ready Target Prompt
                  </span>
                  <Button 
                    variant="ghost" size="icon" className="h-7 w-7 opacity-0 group-hover:opacity-100 transition-opacity"
                    onClick={() => copyStringField("prompt", telemetry.creatorReadyPrompt)}
                  >
                    {copiedField === "prompt" ? <Check className="w-3.5 h-3.5 text-emerald-400" /> : <Copy className="w-3.5 h-3.5" />}
                  </Button>
                </div>
                <p className="text-sm font-medium text-foreground bg-black/30 p-3 rounded-lg font-mono border border-border/20 select-all leading-normal">
                  {telemetry.creatorReadyPrompt}
                </p>
              </div>

              {/* Parameter Row 2: Grid Splits */}
              <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                <div className="bg-background/20 border border-border/30 rounded-xl p-4 space-y-1.5">
                  <span className="text-[10px] font-mono uppercase tracking-wider text-muted-foreground font-bold flex items-center gap-1.5">
                    <Layers className="w-3.5 h-3.5" /> Composition Architecture
                  </span>
                  <p className="text-xs text-foreground leading-relaxed">{telemetry.compositionAnalysis}</p>
                </div>
                <div className="bg-background/20 border border-border/30 rounded-xl p-4 space-y-1.5">
                  <span className="text-[10px] font-mono uppercase tracking-wider text-muted-foreground font-bold flex items-center gap-1.5">
                    <Compass className="w-3.5 h-3.5" /> Lighting Vectors
                  </span>
                  <p className="text-xs text-foreground leading-relaxed">{telemetry.lightingAnalysis}</p>
                </div>
              </div>

              {/* Parameter Row 3: Aesthetic/Atmospheric Breakdowns */}
              <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                <div className="bg-background/20 border border-border/30 rounded-xl p-4 space-y-1.5">
                  <span className="text-[10px] font-mono uppercase tracking-wider text-muted-foreground font-bold flex items-center gap-1.5">
                    <Eye className="w-3.5 h-3.5" /> Scenic Narrative Breakdown
                  </span>
                  <p className="text-xs text-foreground leading-relaxed">{telemetry.cinematicBreakdown}</p>
                </div>
                <div className="bg-background/20 border border-border/30 rounded-xl p-4 space-y-1.5">
                  <span className="text-[10px] font-mono uppercase tracking-wider text-muted-foreground font-bold flex items-center gap-1.5">
                    <Smile className="w-3.5 h-3.5" /> Emotional Grading & Mood
                  </span>
                  <p className="text-xs text-foreground leading-relaxed">
                    <span className="text-cyan-400 font-mono font-semibold uppercase block text-[10px] mb-1">{telemetry.emotionalTone}</span>
                    {telemetry.moodAnalysis}
                  </p>
                </div>
              </div>

              {/* Parameter Row 4: Direct Suggested Marketing Content Hooks */}
              <div className="bg-background/20 border border-border/30 rounded-xl p-4 space-y-3">
                <span className="text-[10px] font-mono uppercase tracking-wider text-muted-foreground font-bold flex items-center gap-1.5">
                  <FileText className="w-3.5 h-3.5" /> Derived Channel Caption Suggestion Blocks
                </span>
                <div className="space-y-2">
                  {telemetry.captionSuggestions.map((caption, idx) => (
                    <div key={idx} className="bg-black/20 p-3 rounded-lg text-xs text-foreground border border-border/20 flex justify-between items-center group/item">
                      <span className="pr-4 leading-normal">{caption}</span>
                      <Button 
                        variant="ghost" size="icon" className="h-6 w-6 shrink-0 opacity-0 group-hover/item:opacity-100 transition-opacity"
                        onClick={() => copyStringField(`cap-${idx}`, caption)}
                      >
                        {copiedField === `cap-${idx}` ? <Check className="w-3 h-3 text-emerald-400" /> : <Copy className="w-3 h-3" />}
                      </Button>
                    </div>
                  ))}
                </div>
              </div>

            </div>
          ) : (
            <div className="absolute inset-0 flex flex-col items-center justify-center text-center p-8 font-mono text-xs text-muted-foreground/40">
              <Eye className="w-8 h-8 mb-3 text-muted-foreground/20" />
              <span>[AWAITING ASSET UPLOAD DEPLOYMENT RAW BUFFER]</span>
              <p className="text-[11px] font-sans text-muted-foreground/30 max-w-sm mt-1 leading-normal">
                Once input data is uploaded, the machine telemetry parser unlocks complex lighting vectors and stable prompt compositions here.
              </p>
            </div>
          )}
        </div>
      </div>

    </SceneDecoderLayout>
  );
}
